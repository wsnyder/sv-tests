module parallel_if;
	reg [3:0] counter;
	wire clk;
	wire reset;
	wire enable;
	wire up_en;
	wire down_en;
	always @(posedge clk)
		if ((reset == 1'b0))
			counter <= 4'b0000;
		else begin
			if (((enable == 1'b1) && (up_en == 1'b1)))
				counter <= (counter + 1'b1);
			if (((enable == 1'b1) && (down_en == 1'b1)))
				counter <= (counter - 1'b1);
		end
endmodule
