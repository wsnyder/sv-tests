	gnt_1

	input req_1;

	wire req_1;

	reg gnt_1;

	parameter GNT0 = 3'b010;
	parameter GNT1 = 3'b100;

			end
			else if ((req_1 == 1'b1)) begin
				fsm_function = GNT1;
			end
			else begin
				fsm_function = IDLE;
			end

				fsm_function = IDLE;
			end

				fsm_function = IDLE;
			end
