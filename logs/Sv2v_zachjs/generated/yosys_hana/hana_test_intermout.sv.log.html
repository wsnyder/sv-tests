module CMP1 (
	in1,
	in2,
	equal,
	unequal,
	greater,
	lesser
);
	parameter SIZE = 1;
	input in1;
	input in2;
	output reg equal;
	output reg unequal;
	output reg greater;
	output reg lesser;
	always @(in1 or in2)
		if ((in1 == in2)) begin
			equal = 1;
			unequal = 0;
			greater = 0;
			lesser = 0;
		end
		else begin
			equal = 0;
			unequal = 1;
			if ((in1 < in2)) begin
				greater = 0;
				lesser = 1;
			end
			else begin
				greater = 1;
				lesser = 0;
			end
		end
endmodule
